```python
def validate_emails(email_list):
	no_valid = '유효하지 않은 이메일 주소입니다.'
	valid = '유효한 이메일 주소입니다.'
	for email in email_list:
		if email.count('@') == 1:
			# 도메인 추출
			domain = email.split('@', 1)[1]
			if domain.count('.') < 1:
				print(email + ' ' + no_valid)
			else:
				print(email + ' ' + valid)
		else:
			print(email + ' ' + no_valid)
```
